 75 function showPGroupagePost(user_groupage_id, start_num, end_num, social_network){
 76 	$.ajax({
 77 		url:'control.PGroupageStream.php',
 78 		data:({
 79 			func:'showPGroupagePost',
 80 			user_groupage_id : user_groupage_id,
 81 			start_num : start_num, 
 82 			end_num : end_num,
 83 			social_network : social_network
 84 		}),
 85 		dataType: 'json',
 86 		async: true,
 87 		success: function(result){
 88 			//alert(result['pic_url']);
 89 			if( result == null ) {
 90 				$("#timeline-content").html("没有结果。");
 91 			} else {
 92 				$("#timeline-content").html("加载中...");
 93 			}
 94 		}
 95 	});
 96 }
